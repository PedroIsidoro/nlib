    Promise.Joint#reject(err) → Void

    Immediately terminate and resolve joint promise with err.

    Example
    joint.reject(Error('OMG|WTF|BBQ'));
 joint.done(function (err, r1, r2) {
  if (err) {
    // err is 'OMG|WTF|BBQ'
  }
   // r1, r2, rN are undefined, as joint was rejected
});
